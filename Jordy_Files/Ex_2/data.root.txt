#include <iostream>
#include <cmath>
#include <cstdlib>
#include "TROOT.h"
#include "TFile.h"
#include "TTree.h"

int main() {
  double mean = 0.0;
  double stddev = 1.0;
  int n = 1000;

  std::random_device rd;
  std::mt19937 gen(rd());
  std::normal_distribution<> dis(mean, stddev);

  std::vector<double> data(n);
  for (int i = 0; i < n; i++) {
    data[i] = dis(gen);
  }

  TFile *file = new TFile("data.root", "w");
  TTree *tree = new TTree("data", file);

  for (int i = 0; i < n; i++) {
    tree->Fill(data[i]);
  }

  file->Close();
  delete file;
  delete tree;

  return 0;
}
